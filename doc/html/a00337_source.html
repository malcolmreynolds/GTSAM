69  template<class FG, class BTCLIQUE>
70  typename JunctionTree<FG,BTCLIQUE>::sharedClique JunctionTree<FG,BTCLIQUE>::distributeFactors(
71  const FG& fg, const SymbolicBayesTree::sharedClique& bayesClique) {
72 
73  // Build "target" index. This is an index for each variable of the factors
74  // that involve this variable as their *lowest-ordered* variable. For each
75  // factor, it is the lowest-ordered variable of that factor that pulls the
76  // factor into elimination, after which all of the information in the
77  // factor is contained in the eliminated factors that are passed up the
78  // tree as elimination continues.
79 
80  // Two stages - first build an array of the lowest-ordered variable in each
81  // factor and find the last variable to be eliminated.
82  std::vector<Index> lowestOrdered(fg.size(), std::numeric_limits<Index>::max());
83  Index maxVar = 0;
84  for(size_t i=0; i<fg.size(); ++i)
85  if(fg[i]) {
86  typename FG::FactorType::const_iterator min = std::min_element(fg[i]->begin(), fg[i]->end());
87  if(min != fg[i]->end()) {
88  lowestOrdered[i] = *min;
89  maxVar = std::max(maxVar, *min);
90  }
91  }
92 
93  // Now add each factor to the list corresponding to its lowest-ordered
94  // variable.
95  std::vector<FastList<size_t> > targets(maxVar+1);
96  for(size_t i=0; i<lowestOrdered.size(); ++i)
97  if(lowestOrdered[i] != std::numeric_limits<Index>::max())
98  targets[lowestOrdered[i]].push_back(i);
99 
100  // Now call the recursive distributeFactors
101  return distributeFactors(fg, targets, bayesClique);
102  }

105  template<class FG, class BTCLIQUE>
106  typename JunctionTree<FG,BTCLIQUE>::sharedClique JunctionTree<FG,BTCLIQUE>::distributeFactors(const FG& fg,
107  const std::vector<FastList<size_t> >& targets,
108  const SymbolicBayesTree::sharedClique& bayesClique) {
109 
110  if(bayesClique) {
111  // create a new clique in the junction tree
112  sharedClique clique(new Clique((*bayesClique)->beginFrontals(), (*bayesClique)->endFrontals(),
113  (*bayesClique)->beginParents(), (*bayesClique)->endParents()));
114 
115  // count the factors for this cluster to pre-allocate space
116  {
117  size_t nFactors = 0;
118  BOOST_FOREACH(const Index frontal, clique->frontal) {
119  // There may be less variables in "targets" than there really are if
120  // some of the highest-numbered variables do not pull in any factors.
121  if(frontal < targets.size())
122  nFactors += targets[frontal].size(); }
123  clique->reserve(nFactors);
124  }
125  // add the factors to this cluster
126  BOOST_FOREACH(const Index frontal, clique->frontal) {
127  if(frontal < targets.size()) {
128  BOOST_FOREACH(const size_t factorI, targets[frontal]) {
129  clique->push_back(fg[factorI]); } } }
130 
131  // recursively call the children
132  BOOST_FOREACH(const typename SymbolicBayesTree::sharedClique bayesChild, bayesClique->children()) {
133  sharedClique child = distributeFactors(fg, targets, bayesChild);
134  clique->addChild(child);
135  child->parent() = clique;
136  }
137  return clique;
138  } else
139  return sharedClique();
140  }

143  template<class FG, class BTCLIQUE>
144  std::pair<typename JunctionTree<FG,BTCLIQUE>::BTClique::shared_ptr,
145  typename FG::sharedFactor> JunctionTree<FG,BTCLIQUE>::eliminateOneClique(
146  typename FG::Eliminate function,
147  const boost::shared_ptr<const Clique>& current) const {
148 
149  FG fg; // factor graph will be assembled from local factors and marginalized children
150  fg.reserve(current->size() + current->children().size());
151  fg.push_back(*current); // add the local factors
152 
153  // receive the factors from the child and its clique point
154  std::list<typename BTClique::shared_ptr> children;
155  BOOST_FOREACH(const boost::shared_ptr<const Clique>& child, current->children()) {
156  std::pair<typename BTClique::shared_ptr, typename FG::sharedFactor> tree_factor(
157  eliminateOneClique(function, child));
158  children.push_back(tree_factor.first);
159  fg.push_back(tree_factor.second);
160  }
161 
162  // eliminate the combined factors
163  // warning: fg is being eliminated in-place and will contain marginal afterwards
164 
165  // Now that we know which factors and variables, and where variables
166  // come from and go to, create and eliminate the new joint factor.
167  gttic(CombineAndEliminate);
168  typename FG::EliminationResult eliminated(function(fg,
169  current->frontal.size()));
170  gttoc(CombineAndEliminate);
171 
172 #ifdef GTSAM_EXTRA_CONSISTENCY_CHECKS
173  assert(std::equal(eliminated.second->begin(), eliminated.second->end(), current->separator.begin()));
174 #endif
175 
176  gttic(Update_tree);
177  // create a new clique corresponding the combined factors
178  typename BTClique::shared_ptr new_clique(BTClique::Create(eliminated));
179  new_clique->children_ = children;
180 
181  BOOST_FOREACH(typename BTClique::shared_ptr& childRoot, children) {
182  childRoot->parent_ = new_clique;
183  }
184  gttoc(Update_tree);
185 
186  return std::make_pair(new_clique, eliminated.second);
187  }

190  template<class FG, class BTCLIQUE>
191  typename BTCLIQUE::shared_ptr JunctionTree<FG,BTCLIQUE>::eliminate(
192  typename FG::Eliminate function) const {
193  if (this->root()) {
194  gttic(JT_eliminate);
195  std::pair<typename BTClique::shared_ptr, typename FG::sharedFactor> ret =
196  this->eliminateOneClique(function, this->root());
197  if (ret.second->size() != 0) throw std::runtime_error(
198  "JuntionTree::eliminate: elimination failed because of factors left over!");
199  gttoc(JT_eliminate);
200  return ret.first;
201  } else
202  return typename BTClique::shared_ptr();
203  }
