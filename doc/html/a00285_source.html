245  boost::optional<Matrix&> H1 = boost::none){

266  inline Point2 project(const Point3& pw,
267  boost::optional<Matrix&> H1 = boost::none,
268  boost::optional<Matrix&> H2 = boost::none,
269  boost::optional<Matrix&> H3 = boost::none) const {
270 
271  if (!H1 && !H2 && !H3) {
272  const Point3 pc = pose_.transform_to(pw) ;
273  if ( pc.z() <= 0 ) throw CheiralityException();
274  const Point2 pn = project_to_camera(pc) ;
275  return K_.uncalibrate(pn);
276  }
277 
278  // world to camera coordinate
279  Matrix Hc1 /* 3*6 */, Hc2 /* 3*3 */ ;
280  const Point3 pc = pose_.transform_to(pw, Hc1, Hc2) ;
281  if( pc.z() <= 0 ) throw CheiralityException();
282 
283  // camera to normalized image coordinate
284  Matrix Hn; // 2*3
285  const Point2 pn = project_to_camera(pc, Hn) ;
286 
287  // uncalibration
288  Matrix Hk, Hi; // 2*2
289  const Point2 pi = K_.uncalibrate(pn, Hk, Hi);
290 
291  // chain the jacobian matrices
292  const Matrix tmp = Hi*Hn ;
293  if (H1) *H1 = tmp * Hc1 ;
294  if (H2) *H2 = tmp * Hc2 ;
295  if (H3) *H3 = Hk;
296  return pi;
297  }

304  inline Point2 project2(const Point3& pw,
305  boost::optional<Matrix&> H1 = boost::none,
306  boost::optional<Matrix&> H2 = boost::none) const {

308  if (!H1 && !H2) {

316  const Point2 pi = this->project(pw, Htmp1, Htmp2, Htmp3);
317  if (H1) {
318  *H1 = Matrix(2, this->dim());
319  H1->leftCols(pose().dim()) = Htmp1 ; // jacobian wrt pose3
320  H1->rightCols(calibration().dim()) = Htmp3 ; // jacobian wrt calib
321  }
322  if (H2) *H2 = Htmp2;
323  return pi;
