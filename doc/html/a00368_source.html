61  template<class FACTORGRAPH>
62  inline void update_dimensions(const FACTORGRAPH& newFactors) {
63  BOOST_FOREACH(const typename FACTORGRAPH::sharedFactor& factor, newFactors) {
64  for(typename FACTORGRAPH::FactorType::const_iterator key = factor->begin(); key != factor->end(); ++key) {
65  if(*key >= dims_.size())
66  dims_.resize(*key + 1);
67  if(dims_[*key] == 0)
68  dims_[*key] = factor->getDim(key);
69  else
70  assert(dims_[*key] == factor->getDim(key));
71  }
72  }
73  }
