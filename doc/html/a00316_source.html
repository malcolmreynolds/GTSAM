37 typename EliminationTree<FACTOR>::sharedFactor EliminationTree<FACTOR>::eliminate_(
38  Eliminate function, Conditionals& conditionals) const {
39 
40  static const bool debug = false;
41 
42  if(debug) std::cout << "ETree: eliminating " << this->key_ << std::endl;
43 
44  if(this->key_ < conditionals.size()) { // If it is requested to eliminate the current variable
45  // Create the list of factors to be eliminated, initially empty, and reserve space
46  FactorGraph<FACTOR> factors;
47  factors.reserve(this->factors_.size() + this->subTrees_.size());
48 
49  // Add all factors associated with the current node
50  factors.push_back(this->factors_.begin(), this->factors_.end());
51 
52  // for all subtrees, eliminate into Bayes net and a separator factor, added to [factors]
53  BOOST_FOREACH(const shared_ptr& child, subTrees_)
54  factors.push_back(child->eliminate_(function, conditionals)); // TODO: spawn thread
55  // TODO: wait for completion of all threads
56 
57  // Combine all factors (from this node and from subtrees) into a joint factor
58  typename FactorGraph<FACTOR>::EliminationResult
59  eliminated(function(factors, 1));
60  conditionals[this->key_] = eliminated.first;
61 
62  if(debug) std::cout << "Eliminated " << this->key_ << " to get:\n";
63  if(debug) eliminated.first->print("Conditional: ");
64  if(debug) eliminated.second->print("Factor: ");
65 
66  return eliminated.second;
67  } else {
68  // Eliminate each child but discard the result.
69  BOOST_FOREACH(const shared_ptr& child, subTrees_) {
70  (void)child->eliminate_(function, conditionals);
71  }
72  return sharedFactor(); // Return a NULL factor
73  }
74 }

77 template<class FACTOR>
78 std::vector<Index> EliminationTree<FACTOR>::ComputeParents(const VariableIndex& structure) {
79 
80  // Number of factors and variables
81  const size_t m = structure.nFactors();
82  const size_t n = structure.size();
83 
84  static const Index none = std::numeric_limits<Index>::max();
85 
86  // Allocate result parent vector and vector of last factor columns
87  std::vector<Index> parents(n, none);
88  std::vector<Index> prevCol(m, none);
89 
90  // for column j \in 1 to n do
91  for (Index j = 0; j < n; j++) {
92  // for row i \in Struct[A*j] do
93  BOOST_FOREACH(const size_t i, structure[j]) {
94  if (prevCol[i] != none) {
95  Index k = prevCol[i];
96  // find root r of the current tree that contains k
97  Index r = k;
98  while (parents[r] != none)
99  r = parents[r];
100  if (r != j) parents[r] = j;
101  }
102  prevCol[i] = j;
103  }
104  }
105 
106  return parents;
107 }

111 template<class DERIVEDFACTOR>
112 typename EliminationTree<FACTOR>::shared_ptr EliminationTree<FACTOR>::Create(
113  const FactorGraph<DERIVEDFACTOR>& factorGraph,
114  const VariableIndex& structure) {
115 
116  static const bool debug = false;
117  gttic(ET_Create1);
118 
119  gttic(ET_ComputeParents);
120  // Compute the tree structure
121  std::vector<Index> parents(ComputeParents(structure));
122  gttoc(ET_ComputeParents);
123 
124  // Number of variables
125  const size_t n = structure.size();
126 
127  static const Index none = std::numeric_limits<Index>::max();
128 
129  // Create tree structure
130  gttic(assemble_tree);
131  std::vector<shared_ptr> trees(n);
132  for (Index k = 1; k <= n; k++) {
133  Index j = n - k; // Start at the last variable and loop down to 0
134  trees[j].reset(new EliminationTree(j)); // Create a new node on this variable
135  if (parents[j] != none) // If this node has a parent, add it to the parent's children
136  trees[parents[j]]->add(trees[j]);
137  else if(!structure[j].empty() && j != n - 1) // If a node other than the last has no parents, this is a forest
138  throw DisconnectedGraphException();
139  }
140  gttoc(assemble_tree);
141 
142  // Hang factors in right places
143  gttic(hang_factors);
144  BOOST_FOREACH(const typename boost::shared_ptr<DERIVEDFACTOR>& factor, factorGraph) {
145  if(factor && factor->size() > 0) {
146  Index j = *std::min_element(factor->begin(), factor->end());
147  if(j < structure.size())
148  trees[j]->add(factor);
149  }
150  }
151  gttoc(hang_factors);
152 
153  if(debug)
154  trees.back()->print("ETree: ");
155 
156  // Check that this is not null
157  assert(trees.back().get());
158  return trees.back();
159 }

188 bool EliminationTree<FACTORGRAPH>::equals(const EliminationTree<FACTORGRAPH>& expected, double tol) const {
189  if(this->key_ == expected.key_ && this->factors_ == expected.factors_
190  && this->subTrees_.size() == expected.subTrees_.size()) {
191  typename SubTrees::const_iterator this_subtree = this->subTrees_.begin();
192  typename SubTrees::const_iterator expected_subtree = expected.subTrees_.begin();
193  while(this_subtree != this->subTrees_.end())
194  if( ! (*(this_subtree++))->equals(**(expected_subtree++), tol))
195  return false;
196  return true;
197  } else
198  return false;
199 }

204  EliminationTree<FACTOR>::eliminatePartial(typename EliminationTree<FACTOR>::Eliminate function, size_t nrToEliminate) const {
205

207  gttic(ET_recursive_eliminate);

211  (void)eliminate_(function, conditionals); // modify in place

217  BOOST_FOREACH(const typename BayesNet::sharedConditional& conditional, conditionals) {
