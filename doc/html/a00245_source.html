98 // Tic and toc functions that are always active (whether or not ENABLE_TIMING is defined)
99 // There is a trick being used here to achieve near-zero runtime overhead, in that a
100 // static variable is created for each tic/toc statement storing an integer ID, but the
101 // integer ID is only looked up by string once when the static variable is initialized
102 // as the program starts.
103 #define gttic_(label) \
104  static const size_t label##_id_tic = ::gtsam::internal::getTicTocID(#label); \
105  ::gtsam::internal::AutoTicToc label##_obj = ::gtsam::internal::AutoTicToc(label##_id_tic, #label)
106 #define gttoc_(label) \
107  label##_obj.stop()
108 #define longtic_(label) \
109  static const size_t label##_id_tic = ::gtsam::internal::getTicTocID(#label); \
110  ::gtsam::internal::ticInternal(label##_id_tic, #label)
111 #define longtoc_(label) \
112  static const size_t label##_id_toc = ::gtsam::internal::getTicTocID(#label); \
113  ::gtsam::internal::tocInternal(label##_id_toc, #label)
