255 template<typename ITERATOR>
256 GaussianConditional::shared_ptr GaussianConditional::Combine(ITERATOR firstConditional, ITERATOR lastConditional) {
257 
258  // TODO: check for being a clique
259 
260  // Get dimensions from first conditional
261  std::vector<size_t> dims; dims.reserve((*firstConditional)->size() + 1);
262  for(const_iterator j = (*firstConditional)->begin(); j != (*firstConditional)->end(); ++j)
263  dims.push_back((*firstConditional)->dim(j));
264  dims.push_back(1);
265 
266  // We assume the conditionals form clique, so the first n variables will be
267  // frontal variables in the new conditional.
268  size_t nFrontals = 0;
269  size_t nRows = 0;
270  for(ITERATOR c = firstConditional; c != lastConditional; ++c) {
271  nRows += dims[nFrontals];
272  ++ nFrontals;
273  }
274 
275  // Allocate combined conditional, has same keys as firstConditional
276  Matrix tempCombined;
277  VerticalBlockView<Matrix> tempBlockView(tempCombined, dims.begin(), dims.end(), 0);
278  GaussianConditional::shared_ptr combinedConditional(new GaussianConditional((*firstConditional)->begin(), (*firstConditional)->end(), nFrontals, tempBlockView, zero(nRows)));
279 
280  // Resize to correct number of rows
281  combinedConditional->matrix_.resize(nRows, combinedConditional->matrix_.cols());
282  combinedConditional->rsd_.rowEnd() = combinedConditional->matrix_.rows();
283 
284  // Copy matrix and sigmas
285  const size_t totalDims = combinedConditional->matrix_.cols();
286  size_t currentSlot = 0;
287  for(ITERATOR c = firstConditional; c != lastConditional; ++c) {
288  const size_t startRow = combinedConditional->rsd_.offset(currentSlot); // Start row is same as start column
289  combinedConditional->rsd_.range(0, currentSlot).block(startRow, 0, dims[currentSlot], combinedConditional->rsd_.offset(currentSlot)).operator=(
290  Matrix::Zero(dims[currentSlot], combinedConditional->rsd_.offset(currentSlot)));
291  combinedConditional->rsd_.range(currentSlot, dims.size()).block(startRow, 0, dims[currentSlot], totalDims - startRow).operator=(
292  (*c)->matrix_);
293  combinedConditional->sigmas_.segment(startRow, dims[currentSlot]) = (*c)->sigmas_;
294  ++ currentSlot;
295  }
296 
297  return combinedConditional;
298 }
