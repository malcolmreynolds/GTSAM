317  boost::shared_ptr<GaussianFactor> linearize(const Values& x, const Ordering& ordering) const {
318  // Only linearize if the factor is active
319  if (!this->active(x))
320  return boost::shared_ptr<JacobianFactor>();
321 
322  // Create the set of terms - Jacobians for each index
323  Vector b;
324  // Call evaluate error to get Jacobians and b vector
325  std::vector<Matrix> A(this->size());
326  b = -unwhitenedError(x, A);
327  if((size_t) b.size() != noiseModel_->dim())
328  throw std::invalid_argument("This factor was created with a NoiseModel of incorrect dimension.");
329 
330  this->noiseModel_->WhitenSystem(A,b);
331 
332  std::vector<std::pair<Index, Matrix> > terms(this->size());
333  // Fill in terms
334  for(size_t j=0; j<this->size(); ++j) {
335  terms[j].first = ordering[this->keys()[j]];
336  terms[j].second.swap(A[j]);
337  }
338 
339  // TODO pass unwhitened + noise model to Gaussian factor
340  noiseModel::Constrained::shared_ptr constrained =
341  boost::dynamic_pointer_cast<noiseModel::Constrained>(this->noiseModel_);
342  if(constrained)
343  return GaussianFactor::shared_ptr(
344  new JacobianFactor(terms, b, constrained->unit()));
345  else
346  return GaussianFactor::shared_ptr(
347  new JacobianFactor(terms, b, noiseModel::Unit::Create(b.size())));
348  }
