459  // Helper function, extracts vectors with variable indices
460  // in the first and last iterators, and concatenates them in that order into the
461  // output.
462  template<typename ITERATOR>
463  const Vector extractVectorValuesSlices(const VectorValues& values, ITERATOR first, ITERATOR last, bool allowNonexistant = false) {
464  // Find total dimensionality
465  size_t dim = 0;
466  for(ITERATOR j = first; j != last; ++j)
467  // If allowNonexistant is true, skip nonexistent indices (otherwise dim will throw an error on nonexistent)
468  if(!allowNonexistant || values.exists(*j))
469  dim += values.dim(*j);
470 
471  // Copy vectors
472  Vector ret(dim);
473  size_t varStart = 0;
474  for(ITERATOR j = first; j != last; ++j) {
475  // If allowNonexistant is true, skip nonexistent indices (otherwise dim will throw an error on nonexistent)
476  if(!allowNonexistant || values.exists(*j)) {
477  ret.segment(varStart, values.dim(*j)) = values[*j];
478  varStart += values.dim(*j);
479  }
480  }
481  return ret;
482  }

485  // Helper function, writes to the variables in values
486  // with indices iterated over by first and last, interpreting vector as the
487  // concatenated vectors to write.
488  template<class VECTOR, typename ITERATOR>
489  void writeVectorValuesSlices(const VECTOR& vector, VectorValues& values, ITERATOR first, ITERATOR last) {
490  // Copy vectors
491  size_t varStart = 0;
492  for(ITERATOR j = first; j != last; ++j) {
493  values[*j] = vector.segment(varStart, values[*j].rows());
494  varStart += values[*j].rows();
