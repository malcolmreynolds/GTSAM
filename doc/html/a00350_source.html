87 VariableSlots::VariableSlots(const FG& factorGraph) {
88  static const bool debug = false;
89 
90  // Compute a mapping (called variableSlots) *from* each involved
91  // variable that will be in the new joint factor *to* the slot in each
92  // removed factor in which that variable appears. For each variable,
93  // this is stored as a vector of slot numbers, stored in order of the
94  // removed factors. The slot number is the max integer value if the
95  // factor does not involve that variable.
96  size_t jointFactorPos = 0;
97  BOOST_FOREACH(const typename FG::sharedFactor& factor, factorGraph) {
98  assert(factor);
99  Index factorVarSlot = 0;
100  BOOST_FOREACH(const Index involvedVariable, *factor) {
101  // Set the slot in this factor for this variable. If the
102  // variable was not already discovered, create an array for it
103  // that we'll fill with the slot indices for each factor that
104  // we're combining. Initially we put the max integer value in
105  // the array entry for each factor that will indicate the factor
106  // does not involve the variable.
107  iterator thisVarSlots; bool inserted;
108  boost::tie(thisVarSlots, inserted) = this->insert(make_pair(involvedVariable, std::vector<Index>()));
109  if(inserted)
110  thisVarSlots->second.resize(factorGraph.size(), std::numeric_limits<Index>::max());
111  thisVarSlots->second[jointFactorPos] = factorVarSlot;
112  if(debug) std::cout << " var " << involvedVariable << " rowblock " << jointFactorPos << " comes from factor's slot " << factorVarSlot << std::endl;
113  ++ factorVarSlot;
114  }
115  ++ jointFactorPos;
116  }
117 }
