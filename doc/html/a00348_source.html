177 template<class FG>
178 void VariableIndex::fill(const FG& factorGraph) {
179  gttic(VariableIndex_fill);
180  // Build index mapping from variable id to factor index
181  for(size_t fi=0; fi<factorGraph.size(); ++fi) {
182  if(factorGraph[fi]) {
183  BOOST_FOREACH(const Index key, factorGraph[fi]->keys()) {
184  if(key < index_.size()) {
185  index_[key].push_back(fi);
186  ++ nEntries_;
187  }
188  }
189  }
190  ++ nFactors_; // Increment factor count even if factors are null, to keep indices consistent
191  }
192 }

195 template<class FG>
196 VariableIndex::VariableIndex(const FG& factorGraph) :
197  nFactors_(0), nEntries_(0)
198 {
199  gttic(VariableIndex_constructor);
200  // If the factor graph is empty, return an empty index because inside this
201  // if block we assume at least one factor.
202  if(factorGraph.size() > 0) {
203  gttic(VariableIndex_constructor_find_highest);
204  // Find highest-numbered variable
205  Index maxVar = 0;
206  BOOST_FOREACH(const typename FG::sharedFactor& factor, factorGraph) {
207  if(factor) {
208  BOOST_FOREACH(const Index key, factor->keys()) {
209  if(key > maxVar)
210  maxVar = key;
211  }
212  }
213  }
214  gttoc(VariableIndex_constructor_find_highest);
215 
216  // Allocate array
217  gttic(VariableIndex_constructor_allocate);
218  index_.resize(maxVar+1);
219  gttoc(VariableIndex_constructor_allocate);
220 
221  fill(factorGraph);
222  }
223 }

237 template<class FG>
238 void VariableIndex::augment(const FG& factors) {
239  gttic(VariableIndex_augment);
240  // If the factor graph is empty, return an empty index because inside this
241  // if block we assume at least one factor.
242  if(factors.size() > 0) {
243  // Find highest-numbered variable
244  Index maxVar = 0;
245  BOOST_FOREACH(const typename FG::sharedFactor& factor, factors) {
246  if(factor) {
247  BOOST_FOREACH(const Index key, factor->keys()) {
248  if(key > maxVar)
249  maxVar = key;
250  }
251  }
252  }
253 
254  // Allocate index
255  index_.resize(std::max(index_.size(), maxVar+1));
256 
257  // Augment index mapping from variable id to factor index
258  size_t orignFactors = nFactors_;
259  for(size_t fi=0; fi<factors.size(); ++fi) {
260  if(factors[fi]) {
261  BOOST_FOREACH(const Index key, factors[fi]->keys()) {
262  index_[key].push_back(orignFactors + fi);
263  ++ nEntries_;
264  }
265  }
266  ++ nFactors_; // Increment factor count even if factors are null, to keep indices consistent
267  }
268  }
269 }

272 template<typename CONTAINER, class FG>
273 void VariableIndex::remove(const CONTAINER& indices, const FG& factors) {
274  gttic(VariableIndex_remove);
275  // NOTE: We intentionally do not decrement nFactors_ because the factor
276  // indices need to remain consistent. Removing factors from a factor graph
277  // does not shift the indices of other factors. Also, we keep nFactors_
278  // one greater than the highest-numbered factor referenced in a VariableIndex.
279  for(size_t fi=0; fi<factors.size(); ++fi)
280  if(factors[fi]) {
281  for(size_t ji = 0; ji < factors[fi]->keys().size(); ++ji) {
282  Factors& factorEntries = index_[factors[fi]->keys()[ji]];
283  Factors::iterator entry = std::find(factorEntries.begin(), factorEntries.end(), indices[fi]);
284  if(entry == factorEntries.end())
285  throw std::invalid_argument("Internal error, indices and factors passed into VariableIndex::remove are not consistent with the existing variable index");
286  factorEntries.erase(entry);
287  -- nEntries_;
288  }
289  }
