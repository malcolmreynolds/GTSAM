356 
358  ISAM2Clique(const sharedConditional& conditional) : Base(conditional) {
359  throw std::runtime_error("ISAM2Clique should always be constructed with the elimination result constructor"); }
360 
362  ISAM2Clique(const std::pair<sharedConditional, boost::shared_ptr<ConditionalType::FactorType> >& result) :
363  Base(result.first), cachedFactor_(result.second),
364  gradientContribution_(result.first->get_R().cols() + result.first->get_S().cols()) {
365  // Compute gradient contribution
366  const ConditionalType& conditional(*result.first);
367  // Rewrite -(R * P')'*d as -(d' * R * P')' for computational speed reasons
368  gradientContribution_ << -conditional.get_R().transpose() * conditional.get_d(),
369  -conditional.get_S().transpose() * conditional.get_d();
370  }
371 
373  shared_ptr clone() const {
374  shared_ptr copy(new ISAM2Clique(std::make_pair(
375  sharedConditional(new ConditionalType(*Base::conditional_)),
376  cachedFactor_ ? cachedFactor_->clone() : Base::FactorType::shared_ptr())));
377  copy->gradientContribution_ = gradientContribution_;
378  return copy;
379  }
380 
382  Base::FactorType::shared_ptr& cachedFactor() { return cachedFactor_; }
383 
385  const Vector& gradientContribution() const { return gradientContribution_; }
386 
387  bool equals(const This& other, double tol=1e-9) const {
388  return Base::equals(other) &&
389  ((!cachedFactor_ && !other.cachedFactor_)
390  || (cachedFactor_ && other.cachedFactor_
391  && cachedFactor_->equals(*other.cachedFactor_, tol)));
392  }
