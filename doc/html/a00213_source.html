224  template<class RHS>
225  void copyStructureFrom(const RHS& rhs) {
226  if((size_t) matrix_.rows() != (size_t) rhs.rows() || (size_t) matrix_.cols() != (size_t) rhs.cols())
227  matrix_.resize(rhs.rows(), rhs.cols());
228  if(rhs.blockStart_ == 0)
229  variableColOffsets_ = rhs.variableColOffsets_;
230  else {
231  variableColOffsets_.resize(rhs.nBlocks() + 1);
232  variableColOffsets_[0] = 0;
233  size_t j=0;
234  assert(rhs.variableColOffsets_.begin()+rhs.blockStart_ < rhs.variableColOffsets_.end()-1);
235  for(std::vector<size_t>::const_iterator off=rhs.variableColOffsets_.begin()+rhs.blockStart_; off!=rhs.variableColOffsets_.end()-1; ++off) {
236  variableColOffsets_[j+1] = variableColOffsets_[j] + (*(off+1) - *off);
237  ++ j;
238  }
239  }
240  rowStart_ = 0;
241  rowEnd_ = matrix_.rows();
242  blockStart_ = 0;
243  assertInvariants();
244  }
