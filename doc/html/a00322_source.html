65  template<class FACTOR, class JUNCTIONTREE>
66  typename BayesTree<typename FACTOR::ConditionalType>::shared_ptr
67  GenericMultifrontalSolver<FACTOR, JUNCTIONTREE>::eliminate(Eliminate function) const {
68 
69  // eliminate junction tree, returns pointer to root
70  typename BayesTree<typename FACTOR::ConditionalType>::sharedClique
71  root = junctionTree_->eliminate(function);
72 
73  // create an empty Bayes tree and insert root clique
74  typename BayesTree<typename FACTOR::ConditionalType>::shared_ptr
75  bayesTree(new BayesTree<typename FACTOR::ConditionalType>);
76  bayesTree->insert(root);
77 
78  // return the Bayes tree
79  return bayesTree;
