274  void BayesTree<CONDITIONAL,CLIQUE>::recursiveTreeBuild(const boost::shared_ptr<BayesTreeClique<IndexConditional> >& symbolic,
275  const std::vector<boost::shared_ptr<CONDITIONAL> >& conditionals,
276  const typename BayesTree<CONDITIONAL,CLIQUE>::sharedClique& parent) {
277 
278  // Helper function to build a non-symbolic tree (e.g. Gaussian) using a
279  // symbolic tree, used in the BT(BN) constructor.
280 
281  // Build the current clique
282  FastList<typename CONDITIONAL::shared_ptr> cliqueConditionals;

470  template<class CONDITIONAL, class CLIQUE>
471  void BayesTree<CONDITIONAL,CLIQUE>::insert(const sharedClique& subtree) {
472  if(subtree) {
473  // Find the parent clique of the new subtree. By the running intersection
474  // property, those separator variables in the subtree that are ordered
475  // lower than the highest frontal variable of the subtree root will all
476  // appear in the separator of the subtree root.
477  if(subtree->conditional()->parents().empty()) {
478  assert(!root_);
479  root_ = subtree;
480  } else {
481  Index parentRepresentative = findParentClique(subtree->conditional()->parents());
482  sharedClique parent = (*this)[parentRepresentative];
483  parent->children_ += subtree;
484  subtree->parent_ = parent; // set new parent!
485  }
486 
487  // Now fill in the nodes index
488  if(nodes_.size() == 0 ||
489  *std::max_element(subtree->conditional()->beginFrontals(), subtree->conditional()->endFrontals()) > (nodes_.size() - 1)) {
490  nodes_.resize(subtree->conditional()->lastFrontalKey() + 1);
491  }
492  fillNodesIndex(subtree);
493  }
494  }

506  sharedClique clique = (*this)[j];

510  FactorGraph<FactorType> cliqueMarginal = clique->marginal(root_,function);
511 #else

528  gttic(Undo_Reduce);
529  result->permuteWithInverse(reduction);
530  BOOST_FOREACH(const boost::shared_ptr<FactorType>& factor, cliqueMarginal) {

532  gttoc(Undo_Reduce);

570  template<class CONDITIONAL, class CLIQUE>

587  }
588  } {
589  sharedClique p = C2;
590  while(p) {
591  path2.push_front(p);
592  p = p->parent();
593  }
594  }
595  // Find the path intersection
596  B = this->root();
597  typename FastList<sharedClique>::const_iterator p1 = path1.begin(), p2 = path2.begin();
598  while(p1 != path1.end() && p2 != path2.end() && *p1 == *p2) {
599  B = *p1;
600  ++p1;
601  ++p2;
602  }
603  }
604  gttoc(Lowest_common_ancestor);
605 
606  // Compute marginal on lowest common ancestor clique
607  gttic(LCA_marginal);
608  FactorGraph<FactorType> p_B = B->marginal2(this->root(), function);
609  gttoc(LCA_marginal);
610 
611  // Compute shortcuts of the requested cliques given the lowest common ancestor
612  gttic(Clique_shortcuts);
613  BayesNet<CONDITIONAL> p_C1_Bred = C1->shortcut(B, function);
614  BayesNet<CONDITIONAL> p_C2_Bred = C2->shortcut(B, function);
615  gttoc(Clique_shortcuts);
616 
617  // Factor the shortcuts to be conditioned on the full root
618  // Get the set of variables to eliminate, which is C1\B.
619  gttic(Full_root_factoring);
620  sharedConditional p_C1_B; {
621  std::vector<Index> C1_minus_B; {
622  FastSet<Index> C1_minus_B_set(C1->conditional()->beginParents(), C1->conditional()->endParents());
623  BOOST_FOREACH(const Index j, *B->conditional()) {
624  C1_minus_B_set.erase(j); }
625  C1_minus_B.assign(C1_minus_B_set.begin(), C1_minus_B_set.end());
626  }
627  // Factor into C1\B | B.
628  FactorGraph<FactorType> temp_remaining;
629  boost::tie(p_C1_B, temp_remaining) = FactorGraph<FactorType>(p_C1_Bred).eliminate(C1_minus_B, function);
630  }
631  sharedConditional p_C2_B; {
632  std::vector<Index> C2_minus_B; {
633  FastSet<Index> C2_minus_B_set(C2->conditional()->beginParents(), C2->conditional()->endParents());
634  BOOST_FOREACH(const Index j, *B->conditional()) {
635  C2_minus_B_set.erase(j); }
636  C2_minus_B.assign(C2_minus_B_set.begin(), C2_minus_B_set.end());
637  }
638  // Factor into C2\B | B.
639  FactorGraph<FactorType> temp_remaining;
640  boost::tie(p_C2_B, temp_remaining) = FactorGraph<FactorType>(p_C2_Bred).eliminate(C2_minus_B, function);
641  }
642  gttoc(Full_root_factoring);
643 
644  gttic(Variable_joint);
645  // Build joint on all involved variables
646  FactorGraph<FactorType> p_BC1C2;
647  p_BC1C2.push_back(p_B);
648  p_BC1C2.push_back(p_C1_B->toFactor());
649  p_BC1C2.push_back(p_C2_B->toFactor());
650  if(C1 != B)
651  p_BC1C2.push_back(C1->conditional()->toFactor());
652  if(C2 != B)
653  p_BC1C2.push_back(C2->conditional()->toFactor());
654 
655  // Reduce the variable indices to start at zero
656  gttic(Reduce);
657  const Permutation reduction = internal::createReducingPermutation(p_BC1C2.keys());
658  internal::Reduction inverseReduction = internal::Reduction::CreateAsInverse(reduction);
659  BOOST_FOREACH(const boost::shared_ptr<FactorType>& factor, p_BC1C2) {
660  if(factor) factor->reduceWithInverse(inverseReduction); }
661  std::vector<Index> js; js.push_back(inverseReduction[j1]); js.push_back(inverseReduction[j2]);
662  gttoc(Reduce);
663 
664  // now, marginalize out everything that is not variable j
665  GenericSequentialSolver<FactorType> solver(p_BC1C2);
666  boost::shared_ptr<FactorGraph<FactorType> > result = solver.jointFactorGraph(js, function);
667 
668  // Undo the reduction
669  gttic(Undo_Reduce);
670  BOOST_FOREACH(const boost::shared_ptr<FactorType>& factor, *result) {
671  if(factor) factor->permuteWithInverse(reduction); }
672  BOOST_FOREACH(const boost::shared_ptr<FactorType>& factor, p_BC1C2) {
673  if(factor) factor->permuteWithInverse(reduction); }
674  gttoc(Undo_Reduce);
675  return result;
676 
677  }

698  template<class CONDITIONAL, class CLIQUE>
699  void BayesTree<CONDITIONAL,CLIQUE>::removePath(sharedClique clique,
700  BayesNet<CONDITIONAL>& bn, typename BayesTree<CONDITIONAL,CLIQUE>::Cliques& orphans) {
701 
702  // base case is NULL, if so we do nothing and return empties above
703  if (clique!=NULL) {
704 
705  // remove the clique from orphans in case it has been added earlier
706  orphans.remove(clique);
707 
708  // remove me
709  this->removeClique(clique);
710 
711  // remove path above me
712  this->removePath(typename Clique::shared_ptr(clique->parent_.lock()), bn, orphans);
713 
714  // add children to list of orphans (splice also removed them from clique->children_)
715  orphans.splice(orphans.begin(), clique->children_);
716 
717  bn.push_back(clique->conditional());
718 
719  }
720  }

730

734  if(clique) {

737  }
738  }
739  }

742  //TODO: Consider Improving

748  template<class CONDITIONAL, class CLIQUE>
749  typename BayesTree<CONDITIONAL,CLIQUE>::Cliques BayesTree<CONDITIONAL,CLIQUE>::removeSubtree(
750  const sharedClique& subtree)
751  {
752  // Result clique list
753  Cliques cliques;
754  cliques.push_back(subtree);
755 
756  // Remove the first clique from its parents
757  if(!subtree->isRoot())
758  subtree->parent()->children().remove(subtree);
759  else
760  root_.reset();
761 
762  // Add all subtree cliques and erase the children and parent of each
763  for(typename Cliques::iterator clique = cliques.begin(); clique != cliques.end(); ++clique)
764  {
765  // Add children
766  BOOST_FOREACH(const sharedClique& child, (*clique)->children()) {
767  cliques.push_back(child); }
768 
769  // Delete cached shortcuts
770  (*clique)->deleteCachedShortcutsNonRecursive();
771 
772  // Remove this node from the nodes index
773  BOOST_FOREACH(Index j, (*clique)->conditional()->frontals()) {
774  nodes_[j].reset(); }
775 
776  // Erase the parent and children pointers
777  (*clique)->parent_.reset();
778  (*clique)->children_.clear();
779  }
780 
781  return cliques;
782  }
