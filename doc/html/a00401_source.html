146 template<class M, class F, class VALUES>
147 typename DoglegOptimizerImpl::IterationResult DoglegOptimizerImpl::Iterate(
148  double Delta, TrustRegionAdaptationMode mode, const M& Rd,
149  const F& f, const VALUES& x0, const Ordering& ordering, const double f_error, const bool verbose) {
150 
151  // Compute steepest descent and Newton's method points
152  gttic(optimizeGradientSearch);
153  gttic(allocateVectorValues);
154  VectorValues dx_u = *allocateVectorValues(Rd);
155  gttoc(allocateVectorValues);
156  gttic(optimizeGradientSearchInPlace);
157  optimizeGradientSearchInPlace(Rd, dx_u);
158  gttoc(optimizeGradientSearchInPlace);
159  gttoc(optimizeGradientSearch);
160  gttic(optimizeInPlace);
161  VectorValues dx_n(VectorValues::SameStructure(dx_u));
162  optimizeInPlace(Rd, dx_n);
163  gttoc(optimizeInPlace);
164  gttic(jfg_error);
165  const GaussianFactorGraph jfg(Rd);
166  const double M_error = jfg.error(VectorValues::Zero(dx_u));
167  gttoc(jfg_error);
168 
169  // Result to return
170  IterationResult result;
171 
172  bool stay = true;
173  enum { NONE, INCREASED_DELTA, DECREASED_DELTA } lastAction = NONE; // Used to prevent alternating between increasing and decreasing in one iteration
174  while(stay) {
175  gttic(Dog_leg_point);
176  // Compute dog leg point
177  result.dx_d = ComputeDoglegPoint(Delta, dx_u, dx_n, verbose);
178  gttoc(Dog_leg_point);
179 
180  if(verbose) std::cout << "Delta = " << Delta << ", dx_d_norm = " << result.dx_d.norm() << std::endl;
181 
182  gttic(retract);
183  // Compute expmapped solution
184  const VALUES x_d(x0.retract(result.dx_d, ordering));
185  gttoc(retract);
186 
187  gttic(decrease_in_f);
188  // Compute decrease in f
189  result.f_error = f.error(x_d);
190  gttoc(decrease_in_f);
191 
192  gttic(decrease_in_M);
193  // Compute decrease in M
194  const double new_M_error = jfg.error(result.dx_d);
195  gttoc(decrease_in_M);
196 
197  if(verbose) std::cout << std::setprecision(15) << "f error: " << f_error << " -> " << result.f_error << std::endl;
198  if(verbose) std::cout << std::setprecision(15) << "M error: " << M_error << " -> " << new_M_error << std::endl;
199 
200  gttic(adjust_Delta);
201  // Compute gain ratio. Here we take advantage of the invariant that the
202  // Bayes' net error at zero is equal to the nonlinear error
203  const double rho = fabs(f_error - result.f_error) < 1e-15 || fabs(M_error - new_M_error) < 1e-15 ?
204  0.5 :
205  (f_error - result.f_error) / (M_error - new_M_error);
206 
207  if(verbose) std::cout << std::setprecision(15) << "rho = " << rho << std::endl;
208 
209  if(rho >= 0.75) {
210  // M agrees very well with f, so try to increase lambda
211  const double dx_d_norm = result.dx_d.norm();
212  const double newDelta = std::max(Delta, 3.0 * dx_d_norm); // Compute new Delta
213 
214  if(mode == ONE_STEP_PER_ITERATION || mode == SEARCH_REDUCE_ONLY)
215  stay = false; // If not searching, just return with the new Delta
216  else if(mode == SEARCH_EACH_ITERATION) {
217  if(newDelta == Delta || lastAction == DECREASED_DELTA)
218  stay = false; // Searching, but Newton's solution is within trust region so keep the same trust region
219  else {
220  stay = true; // Searching and increased Delta, so try again to increase Delta
221  lastAction = INCREASED_DELTA;
222  }
223  } else {
224  assert(false); }
225 
226  Delta = newDelta; // Update Delta from new Delta
227 
228  } else if(0.75 > rho && rho >= 0.25) {
229  // M agrees so-so with f, keep the same Delta
230  stay = false;
231 
232  } else if(0.25 > rho && rho >= 0.0) {
233  // M does not agree well with f, decrease Delta until it does
234  double newDelta;
235  bool hitMinimumDelta;
236  if(Delta > 1e-5) {
237  newDelta = 0.5 * Delta;
238  hitMinimumDelta = false;
239  } else {
240  newDelta = Delta;
241  hitMinimumDelta = true;
242  }
243  if(mode == ONE_STEP_PER_ITERATION || /* mode == SEARCH_EACH_ITERATION && */ lastAction == INCREASED_DELTA || hitMinimumDelta)
244  stay = false; // If not searching, just return with the new smaller delta
245  else if(mode == SEARCH_EACH_ITERATION || mode == SEARCH_REDUCE_ONLY) {
246  stay = true;
247  lastAction = DECREASED_DELTA;
248  } else {
249  assert(false); }
250 
251  Delta = newDelta; // Update Delta from new Delta
252 
253  } else {
254  // f actually increased, so keep decreasing Delta until f does not decrease.
255  // NOTE: NaN and Inf solutions also will fall into this case, so that we
256  // decrease Delta if the solution becomes undetermined.
257  assert(0.0 > rho);
258  if(Delta > 1e-5) {
259  Delta *= 0.5;
260  stay = true;
261  lastAction = DECREASED_DELTA;
262  } else {
263  if(verbose) std::cout << "Warning: Dog leg stopping because cannot decrease error with minimum Delta" << std::endl;
264  result.dx_d.setZero(); // Set delta to zero - don't allow error to increase
265  result.f_error = f_error;
266  stay = false;
267  }
268  }
269  gttoc(adjust_Delta);
270  }
271 
272  // dx_d and f_error have already been filled in during the loop
273  result.Delta = Delta;
274  return result;
275 }
