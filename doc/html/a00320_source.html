103  template<class FACTOR>
104  std::pair<typename FactorGraph<FACTOR>::sharedConditional, FactorGraph<FACTOR> >
105  FactorGraph<FACTOR>::eliminateFrontals(size_t nFrontals, const Eliminate& eliminate) const
106  {
107  // Build variable index
108  VariableIndex variableIndex(*this);
109 
110  // Find first variable
111  Index firstIndex = 0;
112  while(firstIndex < variableIndex.size() && variableIndex[firstIndex].empty())
113  ++ firstIndex;
114 
115  // Check that number of variables is in bounds
116  if(firstIndex + nFrontals > variableIndex.size())
117  throw std::invalid_argument("Requested to eliminate more frontal variables than exist in the factor graph.");
118 
119  // Get set of involved factors
120  FastSet<size_t> involvedFactorIs;
121  for(Index j = firstIndex; j < firstIndex + nFrontals; ++j) {
122  BOOST_FOREACH(size_t i, variableIndex[j]) {
123  involvedFactorIs.insert(i);
124  }
125  }
126 
127  // Separate factors into involved and remaining
128  FactorGraph<FactorType> involvedFactors;
129  FactorGraph<FactorType> remainingFactors;
130  FastSet<size_t>::const_iterator involvedFactorIsIt = involvedFactorIs.begin();
131  for(size_t i = 0; i < this->size(); ++i) {
132  if(involvedFactorIsIt != involvedFactorIs.end() && *involvedFactorIsIt == i) {
133  // If the current factor is involved, add it to involved and increment involved iterator
134  involvedFactors.push_back((*this)[i]);
135  ++ involvedFactorIsIt;
136  } else {
137  // If not involved, add to remaining
138  remainingFactors.push_back((*this)[i]);
139  }
140  }
141 
142  // Do dense elimination on the involved factors
143  typename FactorGraph<FactorType>::EliminationResult eliminationResult =
144  eliminate(involvedFactors, nFrontals);
145 
146  // Add the remaining factor back into the factor graph
147  remainingFactors.push_back(eliminationResult.second);
148 
149  // Return the eliminated factor and remaining factor graph
150  return std::make_pair(eliminationResult.first, remainingFactors);
151  }

154  template<class FACTOR>
155  std::pair<typename FactorGraph<FACTOR>::sharedConditional, FactorGraph<FACTOR> >
156  FactorGraph<FACTOR>::eliminate(const std::vector<KeyType>& variables, const Eliminate& eliminateFcn,
157  boost::optional<const VariableIndex&> variableIndex_) const
158  {
159  const VariableIndex& variableIndex =
160  variableIndex_ ? *variableIndex_ : VariableIndex(*this);
161 
162  // First find the involved factors
163  FactorGraph<FACTOR> involvedFactors;
164  Index highestInvolvedVariable = 0; // Largest index of the variables in the involved factors
165 
166  // First get the indices of the involved factors, but uniquely in a set
167  FastSet<size_t> involvedFactorIndices;
168  BOOST_FOREACH(Index variable, variables) {
169  involvedFactorIndices.insert(variableIndex[variable].begin(), variableIndex[variable].end()); }
170 
171  // Add the factors themselves to involvedFactors and update largest index
172  involvedFactors.reserve(involvedFactorIndices.size());
173  BOOST_FOREACH(size_t factorIndex, involvedFactorIndices) {
174  const sharedFactor factor = this->at(factorIndex);
175  involvedFactors.push_back(factor); // Add involved factor
176  highestInvolvedVariable = std::max( // Updated largest index
177  highestInvolvedVariable,
178  *std::max_element(factor->begin(), factor->end()));
179  }
180 
181  sharedConditional conditional;
182  sharedFactor remainingFactor;
183  if(involvedFactors.size() > 0) {
184  // Now permute the variables to be eliminated to the front of the ordering
185  Permutation toFront = Permutation::PullToFront(variables, highestInvolvedVariable+1);
186  Permutation toFrontInverse = *toFront.inverse();
187  BOOST_FOREACH(const sharedFactor& factor, involvedFactors) {
188  factor->permuteWithInverse(toFrontInverse); }
189 
190  // Eliminate into conditional and remaining factor
191  EliminationResult eliminated = eliminateFcn(involvedFactors, variables.size());
192  conditional = eliminated.first;
193  remainingFactor = eliminated.second;
194 
195  // Undo the permutation
196  BOOST_FOREACH(const sharedFactor& factor, involvedFactors) {
197  factor->permuteWithInverse(toFront); }
198  conditional->permuteWithInverse(toFront);
199  remainingFactor->permuteWithInverse(toFront);
200  } else {
201  // Eliminate 0 variables
202  EliminationResult eliminated = eliminateFcn(involvedFactors, 0);
203  conditional = eliminated.first;
204  remainingFactor = eliminated.second;
205  }
206 
207  // Build the remaining graph, without the removed factors
208  FactorGraph<FACTOR> remainingGraph;
209  remainingGraph.reserve(this->size() - involvedFactors.size() + 1);
210  FastSet<size_t>::const_iterator involvedFactorIndexIt = involvedFactorIndices.begin();
211  for(size_t i = 0; i < this->size(); ++i) {
212  if(involvedFactorIndexIt != involvedFactorIndices.end() && *involvedFactorIndexIt == i)
213  ++ involvedFactorIndexIt;
214  else
215  remainingGraph.push_back(this->at(i));
216  }
217 
218  // Add the remaining factor if it is not empty.
219  if(remainingFactor->size() != 0)
220  remainingGraph.push_back(remainingFactor);
221 
222  return std::make_pair(conditional, remainingGraph);
223 
224  }

264  void _FactorGraph_BayesTree_adder(
265  std::vector<typename boost::shared_ptr<FACTOR> >& factors_io,

267
