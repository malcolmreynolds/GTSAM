93  template<class FACTOR>
94  typename GenericSequentialSolver<FACTOR>::sharedBayesNet //
95  GenericSequentialSolver<FACTOR>::eliminate(const Permutation& permutation,
96  Eliminate function, boost::optional<size_t> nrToEliminate) const
97  {
98  gttic(GenericSequentialSolver_eliminate);
99  // Create inverse permutation
100  Permutation::shared_ptr permutationInverse(permutation.inverse());
101 
102  // Permute the factors - NOTE that this permutes the original factors, not
103  // copies. Other parts of the code may hold shared_ptr's to these factors so
104  // we must undo the permutation before returning.
105  BOOST_FOREACH(const typename boost::shared_ptr<FACTOR>& factor, *factors_)
106  if (factor)
107  factor->permuteWithInverse(*permutationInverse);
108 
109  // Eliminate using elimination tree provided
110  typename EliminationTree<FACTOR>::shared_ptr etree = EliminationTree<FACTOR>::Create(*factors_);
111  sharedBayesNet bayesNet;
112  if(nrToEliminate)
113  bayesNet = etree->eliminatePartial(function, *nrToEliminate);
114  else
115  bayesNet = etree->eliminate(function);
116 
117  // Undo the permutation on the original factors and on the structure.
118  BOOST_FOREACH(const typename boost::shared_ptr<FACTOR>& factor, *factors_)
119  if (factor)
120  factor->permuteWithInverse(permutation);
121 
122  // Undo the permutation on the conditionals
123  BOOST_FOREACH(const boost::shared_ptr<Conditional>& c, *bayesNet)
124  c->permuteWithInverse(permutation);
125 
126  return bayesNet;
127  }

130  template<class FACTOR>
131  typename GenericSequentialSolver<FACTOR>::sharedBayesNet //
132  GenericSequentialSolver<FACTOR>::conditionalBayesNet(
133  const std::vector<Index>& js, size_t nrFrontals,
134  Eliminate function) const
135  {
136  gttic(GenericSequentialSolver_conditionalBayesNet);
137  // Compute a COLAMD permutation with the marginal variables constrained to the end.
138  // TODO in case of nrFrontals, the order of js has to be respected here !
139  Permutation::shared_ptr permutation(
140  inference::PermutationCOLAMD(*structure_, js, true));
141 
142  // Eliminate only variables J \cup F from P(J,F,S) to get P(F|S)
143  size_t nrVariables = structure_->size();
144  size_t nrMarginalized = nrVariables - js.size();
145  size_t nrToEliminate = nrMarginalized + nrFrontals;
146  sharedBayesNet bayesNet = eliminate(*permutation, function, nrToEliminate);
147  // Get rid of conditionals on variables that we want to marginalize out
148  for (size_t i = 0; i < nrMarginalized; i++)
149  bayesNet->pop_front();
150 
151  return bayesNet;
152  }

155  template<class FACTOR>
156  typename GenericSequentialSolver<FACTOR>::sharedBayesNet //
157  GenericSequentialSolver<FACTOR>::jointBayesNet(const std::vector<Index>& js,
158  Eliminate function) const
159  {
160  gttic(GenericSequentialSolver_jointBayesNet);
161  // Compute a COLAMD permutation with the marginal variables constrained to the end.
162  Permutation::shared_ptr permutation(
163  inference::PermutationCOLAMD(*structure_, js));
164 
165  // Eliminate all variables
166  sharedBayesNet bayesNet = eliminate(*permutation, function);
167 
168  // Get rid of conditionals on variables that we want to marginalize out
169  size_t nrMarginalized = bayesNet->size() - js.size();
170  for (size_t i = 0; i < nrMarginalized; i++)
171  bayesNet->pop_front();
172 
173  return bayesNet;
174  }
