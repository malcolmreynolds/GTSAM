45  {

56  allKeys.begin(), allKeys.end(), back_inserter(keep));

147  /* ************************************************************************* */
148  // The shortcut density is a conditional P(S|R) of the separator of this
149  // clique on the root. We can compute it recursively from the parent shortcut
150  // P(Sp|R) as \int P(Fp|Sp) P(Sp|R), where Fp are the frontal nodes in p
151  /* ************************************************************************* */
152  template<class DERIVED, class CONDITIONAL>
153  BayesNet<CONDITIONAL> BayesTreeCliqueBase<DERIVED, CONDITIONAL>::shortcut(
154  derived_ptr B, Eliminate function) const
155  {
156  gttic(BayesTreeCliqueBase_shortcut);
157 
158  // We only calculate the shortcut when this clique is not B
159  // and when the S\B is not empty
160  std::vector<Index> S_setminus_B = separator_setminus_B(B);
161  if (B.get() != this && !S_setminus_B.empty()) {
162 
163  // Obtain P(Cp||B) = P(Fp|Sp) * P(Sp||B) as a factor graph
164  derived_ptr parent(parent_.lock());
165  gttoc(BayesTreeCliqueBase_shortcut);
166  FactorGraph<FactorType> p_Cp_B(parent->shortcut(B, function)); // P(Sp||B)
167  gttic(BayesTreeCliqueBase_shortcut);
168  p_Cp_B.push_back(parent->conditional()->toFactor()); // P(Fp|Sp)
169 
170  // Determine the variables we want to keepSet, S union B
171  std::vector<Index> keep = shortcut_indices(B, p_Cp_B);
172 
173  // Reduce the variable indices to start at zero
174  gttic(Reduce);
175  const Permutation reduction = internal::createReducingPermutation(p_Cp_B.keys());
176  internal::Reduction inverseReduction = internal::Reduction::CreateAsInverse(reduction);
177  BOOST_FOREACH(const boost::shared_ptr<FactorType>& factor, p_Cp_B) {
178  if(factor) factor->reduceWithInverse(inverseReduction); }
179  inverseReduction.applyInverse(keep);
180  gttoc(Reduce);
181 
182  // Create solver that will marginalize for us
183  GenericSequentialSolver<FactorType> solver(p_Cp_B);
184 
185  // Finally, we only want to have S\B variables in the Bayes net, so
186  size_t nrFrontals = S_setminus_B.size();
187  BayesNet<CONDITIONAL> result = *solver.conditionalBayesNet(keep, nrFrontals, function);
188 
189  // Undo the reduction
190  gttic(Undo_Reduce);
191  BOOST_FOREACH(const typename boost::shared_ptr<FactorType>& factor, p_Cp_B) {
192  if (factor) factor->permuteWithInverse(reduction); }
193  result.permuteWithInverse(reduction);
194  gttoc(Undo_Reduce);
195 
196  assertInvariants();
197 
198  return result;
199  } else {
200  return BayesNet<CONDITIONAL>();
201  }
202  }

205  // separator marginal, uses separator marginal of parent recursively
206  // P(C) = P(F|S) P(S)
207  /* ************************************************************************* */
208  template<class DERIVED, class CONDITIONAL>
209  FactorGraph<typename BayesTreeCliqueBase<DERIVED, CONDITIONAL>::FactorType> BayesTreeCliqueBase<
210  DERIVED, CONDITIONAL>::separatorMarginal(derived_ptr R, Eliminate function) const
211  {
212  gttic(BayesTreeCliqueBase_separatorMarginal);
213  // Check if the Separator marginal was already calculated
214  if (!cachedSeparatorMarginal_) {
215  gttic(BayesTreeCliqueBase_separatorMarginal_cachemiss);
216  // If this is the root, there is no separator
217  if (R.get() == this) {
218  // we are root, return empty
219  FactorGraph<FactorType> empty;
220  cachedSeparatorMarginal_ = empty;
221  } else {
222  // Obtain P(S) = \int P(Cp) = \int P(Fp|Sp) P(Sp)
223  // initialize P(Cp) with the parent separator marginal
224  derived_ptr parent(parent_.lock());
225  gttoc(BayesTreeCliqueBase_separatorMarginal_cachemiss); // Flatten recursion in timing outline
226  gttoc(BayesTreeCliqueBase_separatorMarginal);
227  FactorGraph<FactorType> p_Cp(parent->separatorMarginal(R, function)); // P(Sp)
228  gttic(BayesTreeCliqueBase_separatorMarginal);
229  gttic(BayesTreeCliqueBase_separatorMarginal_cachemiss);
230  // now add the parent conditional
231  p_Cp.push_back(parent->conditional()->toFactor()); // P(Fp|Sp)
232 
233  // Reduce the variable indices to start at zero
234  gttic(Reduce);
235  const Permutation reduction = internal::createReducingPermutation(p_Cp.keys());
236  internal::Reduction inverseReduction = internal::Reduction::CreateAsInverse(reduction);
237  BOOST_FOREACH(const boost::shared_ptr<FactorType>& factor, p_Cp) {
238  if(factor) factor->reduceWithInverse(inverseReduction); }
239 
240  // The variables we want to keepSet are exactly the ones in S
241  sharedConditional p_F_S = this->conditional();
242  std::vector<Index> indicesS(p_F_S->beginParents(), p_F_S->endParents());
243  inverseReduction.applyInverse(indicesS);
244  gttoc(Reduce);
245 
246  // Create solver that will marginalize for us
247  GenericSequentialSolver<FactorType> solver(p_Cp);
248 
249  cachedSeparatorMarginal_ = *(solver.jointBayesNet(indicesS, function));
250 
251  // Undo the reduction
252  gttic(Undo_Reduce);
253  BOOST_FOREACH(const typename boost::shared_ptr<FactorType>& factor, p_Cp) {
254  if (factor) factor->permuteWithInverse(reduction); }
255  BOOST_FOREACH(const typename boost::shared_ptr<FactorType>& factor, *cachedSeparatorMarginal_) {
256  if (factor) factor->permuteWithInverse(reduction); }
257  gttoc(Undo_Reduce);
258  }
259  } else {
260  gttic(BayesTreeCliqueBase_separatorMarginal_cachehit);
261  }
262 
263  // return the shortcut P(S||B)
264  return *cachedSeparatorMarginal_; // return the cached version
265  }

285  void BayesTreeCliqueBase<DERIVED, CONDITIONAL>::deleteCachedShortcuts() {
286 
287  // When a shortcut is requested, all of the shortcuts between it and the
288  // root are also generated. So, if this clique's cached shortcut is set,
289  // recursively call over all child cliques. Otherwise, it is unnecessary.
290  if (cachedSeparatorMarginal_) {
291  BOOST_FOREACH(derived_ptr& child, children_) {
292  child->deleteCachedShortcuts();
293  }

295  //Delete CachedShortcut for this clique
