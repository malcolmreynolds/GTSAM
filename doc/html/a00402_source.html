30  template<class VALUE>
31  typename ExtendedKalmanFilter<VALUE>::T ExtendedKalmanFilter<VALUE>::solve_(
32  const GaussianFactorGraph& linearFactorGraph, const Ordering& ordering,
33  const Values& linearizationPoints, Key lastKey,
34  JacobianFactor::shared_ptr& newPrior) const {
35 
36  // Extract the Index of the provided last key
37  gtsam::Index lastIndex = ordering.at(lastKey);
38 
39  // Solve the linear factor graph, converting it into a linear Bayes Network
40  // P(x0,x1) = P(x0|x1)*P(x1)
41  GaussianSequentialSolver solver(linearFactorGraph);
42  GaussianBayesNet::shared_ptr linearBayesNet = solver.eliminate();
43 
44  // Extract the current estimate of x1,P1 from the Bayes Network
45  VectorValues result = optimize(*linearBayesNet);
46  T x = linearizationPoints.at<T>(lastKey).retract(result[lastIndex]);
47 
48  // Create a Jacobian Factor from the root node of the produced Bayes Net.
49  // This will act as a prior for the next iteration.
50  // The linearization point of this prior must be moved to the new estimate of x,
51  // and the key/index needs to be reset to 0, the first key in the next iteration.
52  const GaussianConditional::shared_ptr& cg = linearBayesNet->back();
53  assert(cg->nrFrontals() == 1);
54  assert(cg->nrParents() == 0);
55  // TODO: Find a way to create the correct Jacobian Factor in a single pass
56  JacobianFactor tmpPrior = JacobianFactor(*cg);
57  newPrior = JacobianFactor::shared_ptr(
58  new JacobianFactor(
59  0,
60  tmpPrior.getA(tmpPrior.begin()),
61  tmpPrior.getb()
62  - tmpPrior.getA(tmpPrior.begin()) * result[lastIndex],
63  tmpPrior.get_model()));
64 
65  return x;
66  }

69  template<class VALUE>
70  ExtendedKalmanFilter<VALUE>::ExtendedKalmanFilter(T x_initial,
71  noiseModel::Gaussian::shared_ptr P_initial) {
72 
73  // Set the initial linearization point to the provided mean
74  x_ = x_initial;
75 
76  // Create a Jacobian Prior Factor directly P_initial.
77  // Since x0 is set to the provided mean, the b vector in the prior will be zero
78  // TODO Frank asks: is there a reason why noiseModel is not simply P_initial ?
79  priorFactor_ = JacobianFactor::shared_ptr(
80  new JacobianFactor(0, P_initial->R(), Vector::Zero(x_initial.dim()),
81  noiseModel::Unit::Create(P_initial->dim())));
82  }

85  template<class VALUE>
86  typename ExtendedKalmanFilter<VALUE>::T ExtendedKalmanFilter<VALUE>::predict(
87  const MotionFactor& motionFactor) {
88 
89  // TODO: This implementation largely ignores the actual factor symbols.
90  // Calling predict() then update() with drastically
91  // different keys will still compute as if a common key-set was used
92 
93  // Create Keys
94  Key x0 = motionFactor.key1();
95  Key x1 = motionFactor.key2();
96 
97  // Create an elimination ordering
98  Ordering ordering;
99  ordering.insert(x0, 0);
100  ordering.insert(x1, 1);
101 
102  // Create a set of linearization points
103  Values linearizationPoints;
104  linearizationPoints.insert(x0, x_);
105  linearizationPoints.insert(x1, x_); // TODO should this really be x_ ?
106 
107  // Create a Gaussian Factor Graph
108  GaussianFactorGraph linearFactorGraph;
109 
110  // Add in previous posterior as prior on the first state
111  linearFactorGraph.push_back(priorFactor_);
112 
113  // Linearize motion model and add it to the Kalman Filter graph
114  linearFactorGraph.push_back(
115  motionFactor.linearize(linearizationPoints, ordering));
116 
117  // Solve the factor graph and update the current state estimate
118  // and the posterior for the next iteration.
119  x_ = solve_(linearFactorGraph, ordering, linearizationPoints, x1, priorFactor_);
120 
121  return x_;
122  }
