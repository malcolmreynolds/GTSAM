105 bool equal_with_abs_tol(const Eigen::DenseBase<MATRIX>& A, const Eigen::DenseBase<MATRIX>& B, double tol = 1e-9) {
106 
107  const size_t n1 = A.cols(), m1 = A.rows();
108  const size_t n2 = B.cols(), m2 = B.rows();
109 
110  if(m1!=m2 || n1!=n2) return false;
111 
112  for(size_t i=0; i<m1; i++)
113  for(size_t j=0; j<n1; j++) {
114  if(boost::math::isnan(A(i,j)) ^ boost::math::isnan(B(i,j)))
115  return false;
116  else if(fabs(A(i,j) - B(i,j)) > tol)
117  return false;
118  }
119  return true;
120 }
